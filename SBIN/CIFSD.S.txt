* Called once at process creation

CS.INIT			.DO AUTH
				>LDYA L.LIBCRYPT
				>KAPI LoadLib
				bcs .9

				sta hLIBCRYPT
				.FIN

				clc
.9				rts
*--------------------------------------
* Called until exit with CS
* if RUN exits with CC, RUN entered again
*--------------------------------------
CS.RUN			lda hSocket
				bne .8

				jsr CS.RUN.CheckTCPIP
				bcs .9

				jsr CS.RUN.CheckArgs
				bcs .9

				jsr CS.RUN.Connect
				bcs .9

				jsr INIT.Setup
				bcs .9

				>SS
				>PUSHW L.MSG.MOUNTED

				lda #2
				>KAPI ArgV
				>PUSHYA

				lda #3
				>KAPI ArgV
				>PUSHYA

				>PUSHBI 4
				>LIBC PrintF
				>SR

.8				lda #0
				clc
.9				rts
*--------------------------------------
CS.RUN.CheckTCPIP
				lda #AF.INET
				>STA.G SA.REMOTE

				lda #TCP.PORT.CIFS
				>STA.G SA.REMOTE+S.SOCKADDR.PORT
				iny
				lda /TCP.PORT.CIFS
				sta (pData),y

				clc
				rts
*--------------------------------------
CS.RUN.CheckArgs
				ldy #S.PS.ARGC
				lda (pPS),y
				cmp #3
				bcs .1

				>LDYA L.MSG.USAGE
				>LIBC PutS
				lda #E.SYN
				sec
				rts

.1				lda #TIMEOUT.MAX
				sta TimeOut

				lda #1
				>KAPI ArgV
				bcs .9

				>STYA R1

.2				>SLEEP

				>SS
				>PUSHW R1
				>PUSHWZ
				>PUSHWZ
				>PUSHEA.G pAI
				>LIBC GetAddrInfo
				>SR
				bcc .3

				>SLEEP
				lda TimeOut
				bne .2

				cmp #GAI.E.AGAIN
				bne .3

				lda TimeOut
				bne .2

.3				>SS
				>PUSHW L.MSG.UNKNOWN
				>PUSHW R1
				>PUSHBI 2
				>LIBC PrintF
				>SR

				lda #ERR.SKT.NOCONN
				sec
				rts

.3				lda #4
				>KAPI ArgV
				bcc .4



.4

.8				clc
CS.RUN.CheckArgs.RTS
				rts
*--------------------------------------
CS.RUN.Connect	>SS
				>PUSHWI AF_INET
				>PUSHWI SOCK_SEQPKT
				>PUSHWZ
				>LIBC Socket
				>SR
				bcs .9

				>STYA hSocket

				>SS
				>PUSHW hSocket
				>PUSHW L.SA.LOCAL
				>PUSHWI S.SA.IN
				>LIBC Bind
				>SR
				bcs .9

				lda #TIMEOUT.MAX
				sta TimeOut

.3				>SLEEP

				>SS
				>PUSHW hSocket
				>PUSHEA.G SA.REMOTE
				>PUSHWI S.SA.IN
				>LIBC Connect
				>SR
				bcc .8

				cmp #SKT.E.NOCONN
				bne .9

				ldx TimeOut
				bne .1

.9				pha

				>SS
				>PUSHW L.MSG.NOCONN
				>PUSHW R1
				>PUSHBI 2
				>LIBC PrintF
				>SR
				pla
				sec
				rts

.8				>SS
				>PUSHW L.MSG.CONNECTED
				>PUSHW R1
				>PUSHBI 2
				>LIBC PrintF
				>SR
				rts
*--------------------------------------
CS.SIG			lda TimeOut
				beq .9

				dec TimeOut

.9				sec						do not discard TIMER event
				rts
*--------------------------------------
CS.QUIT			lda hMount

				>LIBC UMount

.1				ldy hSocket
				beq .2

				lda hSocket+1

				>LIBC Shutdown

.2				lda hSMBContext
				beq .3

				>LIBC Free

.3				.DO AUTH
				lda hLIBCRYPT
.7				beq .8

				>KAPI UnloadLib
				.FIN

.8				clc
				rts
